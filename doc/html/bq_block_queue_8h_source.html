33template<typename _Type>
34class bqBlockQueue
35{
36 struct BlockNode
37 {
38 _Type* m_data = 0;
39 };
40
41 BlockNode* m_blocks = 0;
42 uint32_t m_numOfBlocks = 0;
43 uint32_t m_blockSize = 0;
44 uint32_t m_putIndex = 0;
45 bqBlockQueue() = delete;
46
47public:
48 bqBlockQueue(uint32_t blockSize, uint32_t numOfBlocks)
49 :
50 m_numOfBlocks(numOfBlocks),
51 m_blockSize(blockSize)
52 {
53 m_blocks = (BlockNode*)bqMemory::malloc(m_numOfBlocks * sizeof(BlockNode));
54 for (size_t i = 0; i < m_numOfBlocks; ++i)
55 {
56 m_blocks[i].m_data = (_Type*)bqMemory::malloc(m_blockSize * sizeof(_Type));
57 for (size_t ei = 0; ei < m_blockSize; ++ei)
58 {
59 new(&m_blocks[i].m_data[ei]) _Type();
60 }
61 }
62 }
63
64 ~bqBlockQueue()
65 {
66 for (size_t i = 0; i < m_numOfBlocks; ++i)
67 {
68 bqMemory::free(m_blocks[i].m_data);
69 }
70 bqMemory::free(m_blocks);
71 }
72
73 void Put(_Type* blockIn, _Type* blockOut)
74 {
75 // Сначала надо передать blockOut
76 for (size_t i = 0; i < m_blockSize; ++i)
77 {
78 blockOut[i] = m_blocks[m_putIndex].m_data[i];
79
80 m_blocks[m_putIndex].m_data[i] = blockIn[i];
81 }
82
83 ++m_putIndex;
84
85 if (m_putIndex == m_numOfBlocks)
86 m_putIndex = 0;
87 }
88
89 uint32_t BlockSize()const { return m_blockSize; }
90
91 _Type* GetCurrentData() { return m_blocks[m_putIndex].m_data; }
92};
