35// Когда с окном что-то происходит, оно будет вызывать эти коллбэки
36// При создании окна мы обязаны послать указатель на коллбэк
37class bqWindowCallback : public bqUserData
38{
39public:
40 bqWindowCallback() {}
41 virtual ~bqWindowCallback() {}
42 BQ_PLACEMENT_ALLOCATOR(bqWindowCallback);
43
44 // Когда переключились к этому окну
45 virtual void OnActivate(bqWindow*) {}
46 // Когда переключились к другому окну
47 virtual void OnDeactivate(bqWindow*) {}
48
49 // После изменении размера
50 virtual void OnSize(bqWindow*) {}
51 // Пока изменяем размер
52 virtual void OnSizing(bqWindow*) {} // when resizing
53
54 // Свернули окно
55 virtual void OnMinimize(bqWindow*) {}
56 // Развернули на весь монитор
57 virtual void OnMaximize(bqWindow*) {}
58 // Восстановили (из Maximized и Minimized в обычное)
59 virtual void OnRestore(bqWindow*) {}
60
61 // Когда окно перерисовывается
62 virtual void OnDraw(bqWindow*) {}
63
64 // Когда двигается
65 virtual void OnMove(bqWindow*) {}
66
67 // Когда кликается кнопка закрытия
68 virtual void OnClose(bqWindow*) {}
69};

71struct bqWindowCommonData
72{
73 // тут хранится указатель на коллбэк
74 bqWindowCallback* m_cb = 0;
75
76 // размер рамки
77 bqPoint m_borderSize;
78
79 // минимальный размер окна
80 bqPoint m_sizeMinimum;
81
82 // текущий размер окна (клиентской области)
83 bqPoint m_sizeCurrent;
84
85 // координата левого верхнего угла относительно монитора
86 bqPoint m_position;
87
88 // окно видимо или нет
89 bool m_isVisible = false;
90
91 // данные специфичные для ОС
92 void* m_implementation = 0;
93
94 // Работает ли полноэкранный режим без рамки
95 bool m_isFullscreen = false;
96};

98// Окно
99class bqWindow : public bqUserData
100{
101 bqWindowCommonData m_data;
102public:
103 BQ_PLACEMENT_ALLOCATOR(bqWindow);
104 bqWindow(bqWindowCallback* cb);
105 ~bqWindow();
106
107 // Установить заголовок
108 void SetTitle(const char*);
109
110 // Скрыть\показать
111 void SetVisible(bool v);
112
113 // Узнать, видимо ли окно.
114 // Можно скипать рисование если окно например свёрнуто.
115 bool IsVisible() { return m_data.m_isVisible; }
116
117 // Развернуть в полный экран
118 void Maximize();
119 // Свернуть
120 void Minimize();
121 // Восстановить оконное состояние (ну типа оконное, не полный экран)
122 void Restore();
123
124 // Установить\убрать рамку
125 void SetBorderless(bool);
126
127 // Установить\убрать возможность изменять размер
128 void SetNoResize(bool);
129
130 // Показать\скрыть кнопку "свернуть окно"
131 void SetNoMinimize(bool);
132
133 // Тут, указатели для того чтобы получить их и сохранить где-то, и
134 // не вызывать эти методы каждый раз, так как в таком случае потребуется
135 // вызывать постоянно.
136 // Получить минимальный размер окна.
137 bqPoint* GetSizeMinimum() { return &m_data.m_sizeMinimum; }
138 // Получить размер рамки
139 bqPoint* GetBorderSize() { return &m_data.m_borderSize; }
140
141 bqPoint* GetCurrentSize() { return &m_data.m_sizeCurrent; }
142
143 // Получить данные
144 bqWindowCommonData* GetData() { return &m_data; }
145
146 // Установить позицию и размер
147 void SetPositionAndSize(int x, int y, int sx, int sy);
148
149 void ToFullscreenMode();
150 void ToWindowMode();
151
152 // получить центр относительно окна (bqWindow)
153 void GetCenter(bqPoint&);
154};
