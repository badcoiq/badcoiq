37// Имя библиотеки_алгоритм, или просто имя если в ней один алгоритм
38// megalib_deflate,
39// megalib_lz77,
40// megalib_lzma,
41// Имеется:
42// - fastlz (LZ77)
43enum class bqCompressorType : uint32_t
44{
45 // m_level:
46 // 1 скорость
47 // 2 сжатие
48 fastlz
49};
50
51// Информация для сжатия и расжатия
52struct bqCompressionInfo
53{
54 bqCompressorType m_compressorType = bqCompressorType::fastlz;
55
56 // Когда надо сжать данные
57 // Установи размер данных и указатель на данные.
58 // после вызова Compress m_sizeCompressed и m_dataCompressed
59 // будут иметь значения
60 uint32_t m_sizeUncompressed = 0;
61 uint8_t* m_dataUncompressed = 0;
62
63 // Когда надо расжать
64 // Устанавливаем размер сжатых данных, указатель на сжатые данные.
65 // Так-же должен знать m_sizeUncompressed и надо выделить память для распакованных
66 // данных в m_dataUncompressed
67 // После вызова Decompress m_dataUncompressed будет иметь расжатые данные
68 uint32_t m_sizeCompressed = 0;
69 uint8_t* m_dataCompressed = 0;
70
71 // Смотри CompressorType. Если значение не верно
72 // будет использовано значение по умолчанию.
73 uint32_t m_level = 2;
74};

76// Структура с описанием файла архива.
77// Путь к файлу и специфичные данные
78struct bqArchiveZipFile
79{
80 bqStringA m_fileName;
81 void* m_implementation = 0;
82};
83
84class bqArchiveSystem
85{
86public:
87 // Добавить файл из которого будет происходить распаковка
88 static bqArchiveZipFile* ZipAdd(const char* zipFile);
89 static void ZipRemove(bqArchiveZipFile*);
90
91 // Получить список файлов в указанном архиве
92 static void GetFileList(bqArchiveZipFile*, bqArray<bqStringA>&);
93
94 // Распаковать файл. Если `a` NULL первый найденный файл будет распакован.
95 // Установи `a` если используешь много zip файлов которые содержат файлы с одинаковым именем.
96 // Функция выделит память используя bqMemory::malloc, и запишет размер в size
97 // Если всё OK то вернётся указатель.
98 // Используй bqMemory::free для освобождения памяти
99 static uint8_t* ZipUnzip(const char* fileInZip, uint32_t* size, bqArchiveZipFile* a = NULL);
100
101 // Установи все данные в `info`. Если сжатие было OK, вернётся true
102 // и m_dataCompressed будет иметь адрес
103 // Надо вызвать bqMemory::free(m_dataCompressed) для освобождения памяти
104 static bool Compress(bqCompressionInfo* info);
105
106 // Подобно методу `Compress`
107 // Выделяй память для m_dataUncompressed самостоятельно.
108 // Ты должен знать m_sizeUncompressed.
109 static bool Decompress(bqCompressionInfo* info);
110
111 // Расжать данные и сунуть всё в массив
112 // Если info.m_dataCompressed NULL (или !info.m_sizeCompressed)
113 // тогда функция попробует копировать данные из info.m_dataUncompressed
114 // Функция выделит память, вставит все данные в массив и освободит память.
115 // info.m_dataUncompressed и info.m_sizeUncompressed будут проигнорированы
116 static bool Decompress(bqCompressionInfo* info, bqArray<uint8_t>& toArray);
117};
