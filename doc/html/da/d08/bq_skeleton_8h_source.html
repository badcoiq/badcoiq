135// Содержит массив имён джоинтов которые будут анимироваться.
136// Почему имён а не указателей?
137// Потому что объект этого класса он один для всех.
138// Скелеты у каждого объекта свой, так как все могут иметь свою уникальную позу.
139// А для хранения анимации достаточно одного объекта.
140// Для воспроизведения анимации нужно будет добавить ещё один класс
141// который будет на основе имён делать поиск указателей на конкретные джоинты.
142//
143// И массив с фреймами.
144// Каждый фрейм должен содержать массив трансформаций.
145// Трансформации на каждый джоинт. Размер этих массивов должен быть
146// равен размеру массива с именами джоинтов.
147//
148// Всё должно быть удобным для использования в программе. Удобно использовать
149// в программе, но не удобно использовать в загрузчике модели. Поэтому
150// нужно будет добавить методы для удобного добавления джоинтов и фреймов.
151class bqSkeletonAnimation
152{
153 bqArray<bqJointBase> m_joints;
154 bqArray<bqSkeletonAnimationFrame*> m_frames; // "кадры"
155public:
156 // При создании объекта надо указать количество джоинтов и количество фреймов.
157 // Должна произойти инициализация чтобы потом не возникли ошибки.
158 // Инициализация такая: заполнить массив с джоинтами, и фреймы со всеми массивами.
159 bqSkeletonAnimation(uint32_t jtNum, uint32_t frNum, const char* name);
160 ~bqSkeletonAnimation();
161 BQ_PLACEMENT_ALLOCATOR(bqSkeletonAnimation);

163 bqJointBase* GetJoint(size_t i) { return &m_joints.m_data[i]; }
164 void SetJoint(bqJointBase* j, size_t i) { m_joints.m_data[i] = *j; }
