94 void SetRotation(const bqQuaternion& q)
95 {
96 T d = q.Length();
97 T s = 2.0f / d;
98 T xs = q.x * s, ys = q.y * s, zs = q.z * s;
99 T wx = q.w * xs, wy = q.w * ys, wz = q.w * zs;
100 T xx = q.x * xs, xy = q.x * ys, xz = q.x * zs;
101 T yy = q.y * ys, yz = q.y * zs, zz = q.z * zs;
102 Set(
103 1.0f - (yy + zz), xy - wz, xz + wy,
104 xy + wz, 1.0f - (xx + zz), yz - wx,
105 xz - wy, yz + wx, 1.0f - (xx + yy));
106 }

176 bool Invert()
177 {
178 bqMatrix4_t<T> mat;
179 //auto ptr = this->Data();
180 for (unsigned column = 0; column < 4; ++column)
181 {
182 // Swap row in case our pivot point is not working
183 auto column_data = m_data[column].Data();
184 if (column_data[column] == 0)
185 {
186 unsigned big = column;
187 for (unsigned row = 0; row < 4; ++row)
188 {
189 auto row_data = m_data[row].Data();
190 auto big_data = m_data[big].Data();
191 if (fabs(row_data[column]) > fabs(big_data[column]))
192 big = row;
193 }
194 // Print this is a singular matrix, return identity ?
195 if (big == column)
196 fprintf(stderr, "Singular matrix\n");
197 // Swap rows
198 else for (unsigned j = 0; j < 4; ++j)
199 {
200 auto big_data = m_data[big].Data();
201 std::swap(column_data[j], big_data[j]);
202
203 auto other_column_data = mat.m_data[column].Data();
204 auto other_big_data = mat.m_data[big].Data();
205 std::swap(other_column_data[j], other_big_data[j]);
206 }
207 }
208
209 // Set each row in the column to 0
210 for (unsigned row = 0; row < 4; ++row)
211 {
212 if (row != column)
213 {
214 auto row_data = m_data[row].Data();
215 T coeff = row_data[column] / column_data[column];
216 if (coeff != 0)
217 {
218 for (unsigned j = 0; j < 4; ++j)
219 {
220 row_data[j] -= coeff * column_data[j];
221
222 auto other_row_data = mat.m_data[row].Data();
223 auto other_column_data = mat.m_data[column].Data();
224 other_row_data[j] -= coeff * other_column_data[j];
225 }
226 // Set the element to 0 for safety
227 row_data[column] = 0;
228 }
229 }
230 }
231 }
232
233 // Set each element of the diagonal to 1
234 for (unsigned row = 0; row < 4; ++row)
235 {
236 for (unsigned column = 0; column < 4; ++column)
237 {
238 auto other_row_data = mat.m_data[row].Data();
239 auto row_data = m_data[row].Data();
240 other_row_data[column] /= row_data[row];
241 }
242 }
243
244 *this = mat;
245 return true;
246 }
