56 class ClipRectangle
57 {
58 public:
60 ClipRectangle()
61 {
62 setClip(0, 0, 0, 0, 1);
63 }
64
66 ClipRectangle(int aX, int aY, int aWidth, int aHeight, int aScale)
67 {
68 setClip(aX, aY, aWidth, aHeight, aScale);
69 }
70
72 void setClip(int aX, int aY, int aWidth, int aHeight, int aScale)
73 {
74 mMinXi = aX;
75 mMinYi = aY * aScale;
76 mMaxXi = aX + aWidth;
77 mMaxYi = (aY + aHeight) * aScale;
78
79 mXOffset = 0.99f / (float)aScale;
80
81 mMinXf = (float)mMinXi + mXOffset;
82 mMaxXf = (float)mMaxXi - mXOffset;
83
84 mMinYf = (float)mMinYi;
85 mMaxYf = (float)mMaxYi;
86 }
87
89 void intersect(const ClipRectangle& aClipRectangle)
90 {
91 if (mMinXi < aClipRectangle.mMinXi)
92 mMinXi = aClipRectangle.mMinXi;
93 if (mMaxXi > aClipRectangle.mMaxXi)
94 mMaxXi = aClipRectangle.mMaxXi;
95 if (mMinXi > mMaxXi)
96 mMinXi = mMaxXi;
97
98 if (mMinYi < aClipRectangle.mMinYi)
99 mMinYi = aClipRectangle.mMinYi;
100 if (mMaxYi > aClipRectangle.mMaxYi)
101 mMaxYi = aClipRectangle.mMaxYi;
102 if (mMinYi > mMaxYi)
103 mMinYi = mMaxYi;
104
105 mMinXf = (float)mMinXi + mXOffset;
106 mMaxXf = (float)mMaxXi - mXOffset;
107
108 mMinYf = (float)mMinYi;
109 mMaxYf = (float)mMaxYi;
110 }
111
113 inline int getMinXi() const { return mMinXi; }
114
116 inline int getMinYi() const { return mMinYi; }
