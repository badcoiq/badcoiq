33// Функция чтобы получить длинну строки
34template<typename other_type>
35size_t bqStringLen(const other_type* str)
36{
37 // Считаем пока не встретится ноль
38 unsigned int len = 0u;
39 if (str[0] != 0)
40 {
41 const other_type* p = &str[0u];
42 while ((size_t)*p++)
43 ++len;
44 }
45 return len;
46}

56 // если количество символов m_size равно или больше m_allocated то надо вызвать это
57 // указав новый m_allocated
58 void _reallocate(size_t new_allocated)
59 {
60 // выделение новой памяти, копирование содержимого, освобождение старой, сохранение нового указателя
61 char_type* new_data = (char_type*)bqMemory::malloc(new_allocated * sizeof(char_type));
62 if (m_data)
63 {
64 memcpy(new_data, m_data, m_size * sizeof(char_type));
65 bqMemory::free(m_data);
66 }
67 else
68 {
69 memset(new_data, 0, new_allocated);
70 }
71 m_data = new_data;
72 m_allocated = new_allocated;
73 }

509 // конвертировать в int
510 int32_t to_int()
511 {
512 size_t len = m_size;
513 int result = 0;
514 int mul_val = 1;
515 bool is_neg = m_data[0] == (char_type)'-';
516 for (size_t i = 0, last = len - 1; i < len; ++i)
517 {
518 int32_t char_value = (int32_t)m_data[last] - 0x30;
519 if (char_value >= 0 && char_value <= 9)
520 {
521 result += char_value * mul_val;
522 mul_val *= 10;
523 }
524 --last;
525 }
526 if (is_neg)
527 result = -result;
528 return result;
529 }

531 // конвертировать в float
532 float to_float()
533 {
534 float result = 0.f;
535 auto ptr = m_data;
536 bool is_negative = *ptr == (char_type)'-';
537
538 if (is_negative)
539 ++ptr;
540
541 int i = 0;
542 while (*ptr >= (char_type)'0' && *ptr <= (char_type)'9')
543 {
544 i *= 10;
545 i += *ptr - (char_type)'0';
546 ++ptr;
547 }
548 result = (float)i;
549
550 i = 0;
551
552 if (*ptr == (char_type)'.')
553 ++ptr;
554
555 int part_2_count = 0;
556 while (*ptr >= (char_type)'0' && *ptr <= (char_type)'9')
557 {
558 i *= 10;
559 i += *ptr - (char_type)'0';
560 ++ptr;
561 ++part_2_count;
562 }

564 const static float string_to_float_table[17] =
565 {
566 0.f,
567 0.1f,

569 0.001f,
570 0.0001f,
571 0.00001f,
572 0.000001f,
573 0.0000001f,
574 0.00000001f,
575 0.000000001f,

577 0.00000000001f,
578 0.000000000001f,
579 0.0000000000001f,

581 0.000000000000001f,
582 0.0000000000000001f,
583 };
584
585 result += (float)i * string_to_float_table[part_2_count];
586
587 return is_negative ? -result : result;

590 // вернуть имя файла без расширения
591 bqString_base<char_type> stem()
592 {
593 bqString_base<char_type> r;
594
595 if (m_size)
596 {
597 bqString_base<char_type> tmp;
598 for (size_t i = m_size - 1; i >= 0; --i)
599 {
600 if ((m_data[i] != (char_type)'/') && (m_data[i] != (char_type)'\\'))
601 tmp.push_back(m_data[i]);
602 else
603 break;
604
605 if (!i)
606 break;
607 }
608
609 tmp.flip();
610 tmp.pop_extension();
611
612 r = tmp;
613 }
614
615 return r;
616 }

618 bool has_extension()
619 {
620 bool r = false;
621
622 for (size_t i = m_size - 1; i >= 0; --i)
623 {
624 // первая точка с конца
625 if (m_data[i] == '.')
626 {
627 // если /.bar
628 // то нет расширения
629 if (i > 0)
630 {
631 if ((m_data[i - 1] == '/') || (m_data[i - 1] == '\\'))
632 break;
633 }
634
635 // если /foo/.bar./
636 // то нет расширения
637 if (i == m_size - 1)
638 break;
639
640 r = true;
641 break;
642 }
643
644 if (!i)
645 break;
646 }
647
648 return r;
649 }
