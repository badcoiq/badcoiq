38template<class _type>
39struct bqResourceCacheNode
40{
41 _type m_resource;
42 bqMD5 m_md5;
43
46 bqStringA* m_path = 0;
47};

51template<class _type>
52class bqResourceCache
53{
54 bqArray<bqResourceCacheNode<_type>*> m_data;
55public:
56 bqResourceCache() { m_data.reserve(100); }
57 virtual ~bqResourceCache() { Clear(); }
58 BQ_PLACEMENT_ALLOCATOR(bqResourceCache);
59
60 virtual void Free(_type r) = 0;
61 virtual _type Load(const char* path) = 0;
62
63 _type Get(const char* path)
64 {
65 _type t = Get(bqCryptography::MD5(path, strlen(path)));
66 if (!t)
67 t = Get(path, 0, true);
68 return t;
69 }
70 _type Get(const bqMD5& md5)
71 {
72 for (size_t i = 0; i < m_data.m_size; ++i)
73 {
74 if (bqCryptography::Compare(m_data.m_data[i]->m_md5, md5))
75 return m_data.m_data[i]->m_resource;
76 }
77 return 0;
78 }
79 _type Get(const char* path, uint32_t* outIndex, bool load = true)
80 {
81 _type t = Get(bqCryptography::MD5(path, strlen(path)));
82 if (!t)
83 {
84 bqResourceCacheNode<_type>* n = new bqResourceCacheNode<_type>;
85 n->m_resource = 0;
86 n->m_md5 = bqCryptography::MD5(path, strlen(path));
87 n->m_path = new bqStringA(path);
88 m_data.push_back(n);
89
90 uint32_t index = m_data.size() - 1;
91
92 if (outIndex)
93 *outIndex = index;
94
95 if (load)
96 t = Reload(index);
97 }
98 return t;
99 }
100 _type Get(uint32_t i)
101 {
102 if (i < m_data.m_size)
103 return m_data.m_data[i]->m_resource;
104 return 0;
105 }
106 uint32_t GetNum()
107 {
108 return (uint32_t)m_data.m_size;
109 }
110 void Clear()
111 {
112 for (size_t i = 0; i < m_data.m_size; ++i)
113 {
114 Free(m_data.m_data[i]->m_resource);
115 m_data.m_data[i]->m_resource = 0;
116
117 if (m_data.m_data[i]->m_path)
118 delete m_data.m_data[i]->m_path;
119 delete m_data.m_data[i];
120 }
121 m_data.clear();
122 }
123 uint32_t GetIndex(const char* path)
124 {
125 return GetIndex(bqCryptography::MD5(path, strlen(path)));
126 }
127 uint32_t GetIndex(const bqMD5& md5)
128 {
129 for (size_t i = 0; i < m_data.m_size; ++i)
130 {
131 if (bqCryptography::Compare(m_data.m_data[i]->m_md5, md5))
132 return (uint32_t)i;
133 }
134 return 0xFFFFFFFF;
135 }
136 uint32_t GetIndex(_type r)
137 {
138 for (size_t i = 0; i < m_data.m_size; ++i)
139 {
140 if (m_data.m_data[i]->m_resource == r)
141 return (uint32_t)i;
142 }
143 return 0xFFFFFFFF;
144 }
145
146 void Unload(uint32_t i)
147 {
148 if (i < m_data.m_size)
149 {
150 Free(m_data.m_data[i]->m_resource);
151 m_data.m_data[i]->m_resource = 0;
152 }
153 }
154
155
156 _type Reload(uint32_t index, bool forceUnload = false)
157 {
158 _type t = 0;
159 if (index < m_data.m_size)
160 {
161 if (forceUnload)
162 Unload(index);
163
164 if (m_data.m_data[index]->m_path
165 && !m_data.m_data[index]->m_resource)
166 {
167 t = Load(m_data.m_data[index]->m_path->c_str());
168 m_data.m_data[index]->m_resource = t;
169 }
170
171 }
172 return t;
173 }
174 bool IsLoaded(uint32_t i)
175 {
176 if (i < m_data.m_size)
177 return (m_data.m_data[i]->m_resource != 0);
178 return false;
179 }
180};
