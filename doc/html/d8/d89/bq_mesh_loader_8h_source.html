38class bqMeshLoaderCallback
39{
40public:
41 bqMeshLoaderCallback() {}
42 virtual ~bqMeshLoaderCallback() {}
43 BQ_PLACEMENT_ALLOCATOR(bqMeshLoaderCallback);
44
45 virtual void OnMaterialLog(bqMaterial* m)
46 {
47 if (m)
48 {
49 bqLog::PrintInfo("Mesh Callback: Material:\n");
50 if (m->m_name.size())
51 bqLog::PrintInfo("\t\tname: %s\n", bqFramework::GetUTF8String(m->m_name).c_str());
52 else
53 bqLog::PrintInfo("\t\tname: without name :(\n");
54 }
55 }
56
57 virtual void OnMeshLog(bqMesh* newMesh, bqString* name, bqString* materialName)
58 {
59 if (newMesh)
60 {
61 bqLog::PrintInfo("Mesh Callback: new mesh:\n");
62 if (name->size())
63 bqLog::PrintInfo("\tname: %s\n", bqFramework::GetUTF8String(*name).c_str());
64 else
65 bqLog::PrintInfo("\tname: without name :(\n");
66
67 if (materialName->size())
68 bqLog::PrintInfo("\tmaterialName: %s\n", bqFramework::GetUTF8String(*materialName).c_str());
69 else
70 bqLog::PrintInfo("\tmaterialName: without name :(\n");
71
72
73 switch (newMesh->GetInfo().m_indexType)
74 {
75 case bqMeshIndexType::u16:
76 bqLog::PrintInfo("\tindex type: u16\n");
77 break;
78 case bqMeshIndexType::u32:
79 bqLog::PrintInfo("\tindex type: u32\n");
80 break;
81 }
82
83 switch (newMesh->GetInfo().m_vertexType)
84 {
85 case bqMeshVertexType::Null:
86 bqLog::PrintInfo("\tvertex type: Null\n");
87 break;
88 case bqMeshVertexType::Triangle:
89 bqLog::PrintInfo("\tvertex type: Triangle\n");
90 break;
91 }
92
93 bqLog::PrintInfo("\tv num: %u\n", newMesh->GetInfo().m_vCount);
94 bqLog::PrintInfo("\ti num: %u\n", newMesh->GetInfo().m_iCount);
95 bqLog::PrintInfo("\tstride: %u\n", newMesh->GetInfo().m_stride);
96
97 bqLog::PrintInfo("\tskinned: %s\n", newMesh->GetInfo().m_skinned ? "true" : "false");
98 bqLog::PrintInfo("\taabb: \n");
99 bqLog::PrintInfo("\t\tminx: [%f] [%f] :maxx\n", newMesh->GetInfo().m_aabb.m_min.x, newMesh->GetInfo().m_aabb.m_max.x);
100 bqLog::PrintInfo("\t\tminy: [%f] [%f] :maxy\n", newMesh->GetInfo().m_aabb.m_min.y, newMesh->GetInfo().m_aabb.m_max.y);
101 bqLog::PrintInfo("\t\tminz: [%f] [%f] :maxz\n", newMesh->GetInfo().m_aabb.m_min.z, newMesh->GetInfo().m_aabb.m_max.z);
102 }
103 }
104
105 virtual void OnMaterial(bqMaterial* m)
106 {
107 OnMaterialLog(m);
108 }
109
110 virtual void OnMesh(bqMesh* newMesh, bqString* n, bqString* mn)
111 {
112 if (newMesh)
113 {
114 OnMeshLog(newMesh, n, mn);
115
116 // if don't need then destroy this mesh
117 bqDestroy(newMesh);
118 }
119 }
120
121 // Будет вызван по окончании чтения файла.
122 // Неизвестно что будет прочитано первым, материал или модель.
123 // После чтения, можно будет в данном методе найти нужный материал для нужной модели.
124 virtual void Finale()
125 {
126
127 }

132 {
133 // if don't need then destroy
134 bqDestroy(s);
135 }
136 }
137 virtual void OnAnimation(bqSkeletonAnimation* a)
138 {
139 if (a)
140 {
141 bqDestroy(a);
142 }
143 }
