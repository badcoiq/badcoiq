66class bqTextBufferReader
67{
68 uint8_t* m_ptrCurr = 0;
69 uint8_t* m_ptrBegin = 0;
70 uint8_t* m_ptrEnd = 0;
71 size_t m_size = 0;
72 void _onConstructor()
73 {
74 m_ptrBegin = m_ptrCurr;
75 m_ptrEnd = m_ptrCurr + m_size;
76 }
77
78 void _skipSpaces() {
79 while (!IsEnd()) {
80 if (!isspace(*m_ptrCurr))
81 break;
82 ++m_ptrCurr;
83 }
84 }
85
86 bool _isdelim(const char* delim, char ch)
87 {
88 if (delim)
89 {
90 const char* ptr = delim;
91 while (*ptr)
92 {
93 if (*ptr == ch)
94 return true;
95
96 ++ptr;
97 }
98 }
99 return false;
100 }

149 // Получить всё между символами isspace()
150 void GetWord(bqStringA& out, const char* delim)
151 {
152 _skipSpaces();
153 out.clear();
154 while (!IsEnd())
155 {
156 if (isspace(*m_ptrCurr) || _isdelim(delim, *m_ptrCurr))
157 break;
158 out.push_back(*m_ptrCurr);
159 ++m_ptrCurr;
160 }

223 // Получить остаток текущей строки
224 void GetLine(bqStringA& out)
225 {
226 out.clear();
227 _skipSpaces();
228
229 while (!IsEnd())
230 {
231 if (*m_ptrCurr == '\r' || *m_ptrCurr == '\n')
232 {
233 SkipLine();
234 break;
235 }
236 out.push_back(*m_ptrCurr);
237 ++m_ptrCurr;
238 }
