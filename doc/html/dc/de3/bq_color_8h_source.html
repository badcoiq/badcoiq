35class bqColor
36{
37public:
38 // RGBA
39 float m_data[4];
40
41 bqColor()
42 {
43 m_data[0] = m_data[1] = m_data[2] = 0.f;
44 m_data[3] = 1.f;
45 }
46
47 bqColor(float v)
48 {
49 m_data[0] = m_data[1] = m_data[2] = v;
50 m_data[3] = 1.f;
51 }
52
53 bqColor(float r, float g, float b, float a = 1.f)
54 {
55 m_data[0] = r;
56 m_data[1] = g;
57 m_data[2] = b;
58 m_data[3] = a;
59 }
60
61 bqColor(int32_t r, int32_t g, int32_t b, int32_t a = 255)
62 {
63 this->SetAsByteAlpha(a);
64 this->SetAsByteRed(r);
65 this->SetAsByteGreen(g);
66 this->SetAsByteBlue(b);
67 }
68
69 bqColor(uint32_t uint_data) // 0xFF112233 ARGB
70 {
71 SetAsInteger(uint_data);
72 }
73
74 bqColor(int32_t int_data) // 0xFF112233 ARGB
75 {
76 SetAsInteger(0xFF000000 | (uint32_t)int_data);
77 }
78
79 float X() { return m_data[0]; }
80 float Y() { return m_data[1]; }
81 float Z() { return m_data[2]; }
82 float W() { return m_data[3]; }
83
84 const float* Data() const { return &m_data[0]; }
85
86 float GetRed() { return m_data[0]; }
87 float GetGreen() { return m_data[1]; }
88 float GetBlue() { return m_data[2]; }
89 float GetAlpha() { return m_data[3]; }
90
91 uint8_t GetAsByteRed() const { return (uint8_t)floor(m_data[0] * 255.); }
92 uint8_t GetAsByteGreen() const { return (uint8_t)floor(m_data[1] * 255.); }
93 uint8_t GetAsByteBlue() const { return (uint8_t)floor(m_data[2] * 255.); }
94 uint8_t GetAsByteAlpha() const { return (uint8_t)floor(m_data[3] * 255.); }
95
96 bool operator==(const bqColor& v) const
97 {
98 if (m_data[0] != v.m_data[0]) return false;
99 if (m_data[1] != v.m_data[1]) return false;
100 if (m_data[2] != v.m_data[2]) return false;
101 if (m_data[3] != v.m_data[3]) return false;
102 return true;
103 }
104
105 bool operator!=(const bqColor& v) const
106 {
107 if (m_data[0] != v.m_data[0]) return true;
108 if (m_data[1] != v.m_data[1]) return true;
109 if (m_data[2] != v.m_data[2]) return true;
110 if (m_data[3] != v.m_data[3]) return true;
111 return false;
112 }
113
114 void Normalize()
115 {
116 float len = sqrtf((m_data[0] * m_data[0]) + (m_data[1] * m_data[1]) + (m_data[2] * m_data[2]));
117 if (len > 0)
118 {
119 len = 1.0f / len;
120 }
121 m_data[0] *= len;
122 m_data[1] *= len;
123 m_data[2] *= len;
124 }
125
126 // 0xff112233
127 uint32_t getAsInteger()
128 {
129 return BQ_MAKEFOURCC(
130 this->GetAsByteBlue(),
131 this->GetAsByteGreen(),
132 this->GetAsByteRed(),
133 this->GetAsByteAlpha()
134 );
135 }
136
137 void SetAlpha(float v) { m_data[3] = v; }
138 void SetRed(float v) { m_data[0] = v; }
139 void SetGreen(float v) { m_data[1] = v; }
140 void SetBlue(float v) { m_data[2] = v; }
141
142 void SetR(float v) { m_data[0] = v; }
143 void SetG(float v) { m_data[1] = v; }
144 void SetB(float v) { m_data[2] = v; }
145 void SetRGB(float r, float g, float b)
146 {
147 m_data[0] = r;
148 m_data[1] = g;
149 m_data[2] = b;
150 }
151
152 void SetAsByteAlpha(int32_t v) { m_data[3] = static_cast<float>(v) * 0.00392156862745f; }
153 void SetAsByteRed(int32_t v) { m_data[0] = static_cast<float>(v) * 0.00392156862745f; }
154 void SetAsByteGreen(int32_t v) { m_data[1] = static_cast<float>(v) * 0.00392156862745f; }
155 void SetAsByteBlue(int32_t v) { m_data[2] = static_cast<float>(v) * 0.00392156862745f; }
156
157 void SetAsInteger(uint32_t v)
158 {
159 this->SetAsByteRed(static_cast<uint8_t>(v >> 16));
160 this->SetAsByteGreen(static_cast<uint8_t>(v >> 8));
161 this->SetAsByteBlue(static_cast<uint8_t>(v));
162 this->SetAsByteAlpha(static_cast<uint8_t>(v >> 24));
163 }
164
165 void Set(float r, float g, float b, float a = 1.)
166 {
167 SetAlpha(a);
168 SetRed(r);
169 SetGreen(g);
170 SetBlue(b);
171 }
172
173 void Set(float v, float a = 1.)
174 {
175 SetAlpha(a);
176 SetRed(v);
177 SetGreen(v);
178 SetBlue(v);
179 }
180
181 void Set(const bqColor& other)
182 {
183 *this = other;
184 }
185
186 bqVec4f GetVec4()
187 {
188 return bqVec4f(m_data[0], m_data[1], m_data[2], m_data[3]);
189 }
190};
