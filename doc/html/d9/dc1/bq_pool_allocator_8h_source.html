20class bqPoolAllocator
21{
22 uint32_t m_elemSize;
23 uint32_t m_maxElements;
24 uint32_t m_freeCount;
25 void* m_firstFree;
26 uint8_t* m_pool;
27
28public:
29 bqPoolAllocator(uint32_t elemSize, uint32_t maxElements)
30 : m_elemSize(elemSize),
31 m_maxElements(maxElements)
32 {
33 BQ_ASSERT_ST(elemSize);
34 BQ_ASSERT_ST(maxElements);
35
36 m_pool = (uint8_t*)bqMemory::malloc(m_elemSize * m_maxElements);
37
38 uint8_t* p = m_pool;
39 m_firstFree = p;
40 m_freeCount = m_maxElements;
41 uint32_t count = m_maxElements;
42 while (--count)
43 {
44 *(void**)p = (p + m_elemSize);
45 p += m_elemSize;
46 }
47 *(void**)p = 0;
48 }
49
50 ~bqPoolAllocator()
51 {
52 bqMemory::free(m_pool);
53 }
54
55 uint32_t GetFreeCount() const
56 {
57 return m_freeCount;
58 }
59
60 uint32_t GetUsedCount() const
61 {
62 return m_maxElements - m_freeCount;
63 }
64
65 uint32_t GetMaxCount() const
66 {
67 return m_maxElements;
68 }
69
70 void* Allocate(uint32_t size)
71 {
72 // release mode fix
73 (void)size;
74 BQ_ASSERT_ST(!size || size <= m_elemSize);
75 BQ_ASSERT_ST(m_freeCount > 0);
76 void* result = m_firstFree;
77 m_firstFree = *(void**)m_firstFree;
78 --m_freeCount;
79 return result;
80 }
81
82 bool ValidPtr(void* ptr)
83 {
84 if (ptr)
85 {
86 if (((uint8_t*)ptr >= m_pool && (uint8_t*)ptr < m_pool + m_maxElements * m_elemSize))
87 {
88 return true;
89 }
90 }
91 return false;
92 }
93
94 void FreeMemory(void* ptr)
95 {
96 if (ptr)
97 {
98 BQ_ASSERT_ST((uint8_t*)ptr >= m_pool && (uint8_t*)ptr < m_pool + m_maxElements * m_elemSize);
99
100 *(void**)ptr = m_firstFree;
101 m_firstFree = ptr;
102 ++m_freeCount;
103 }
104 }
105
106 int GetElementSize() const
107 {
108 return m_elemSize;
109 }
110
111 uint8_t* GetPoolAddress()
112 {
113 return m_pool;
114 }
115
116 const uint8_t* GetPoolAddress() const
117 {
118 return m_pool;
119 }
