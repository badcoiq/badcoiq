147 bqXMLAttribute* getAttribute(const bqStringA& Name)
148 {
149 size_t sz = m_attributeList.size();
150 for (size_t i = 0; i < sz; ++i)
151 {
152 if (m_attributeList[i]->m_name == Name)
153 {
154 return m_attributeList[i];
155 }
156 }
157 return nullptr;
158 }
159
160 void clear()
161 {
162 m_name.clear();
163 m_text.clear();
164 size_t sz = m_attributeList.size();
165 for (size_t i = 0; i < sz; ++i)
166 {
167 delete m_attributeList[i];
168 }
169 sz = m_nodeList.size();
170 for (size_t i = 0; i < sz; ++i)
171 {
172 delete m_nodeList[i];
173 }
174 m_attributeList.clear();
175 m_nodeList.clear();
176 }
