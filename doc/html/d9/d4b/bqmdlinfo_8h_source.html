81// .mdl файл должен содержать всё необходимое для работы.
82// Файл разбит на чанки - куски данных отвечающих за что-то.
83// Каждый чанк содержит свой общий заголовок и заголовок
84// конкретного типа чанка.
85//
86// Начинается файла с главного заголовка.
87struct bqMDLFileHeader
88{
89 // строка "bmdl"
90 uint32_t m_bmld = 1818520930;
91
92 // версия
93 uint32_t m_version = 1;
94
95 // количество чанков
96 uint32_t m_chunkNum = 0;
97
98 // типы алгоритмов сжатия.
99 enum
100 {
101 compression_null,
102
103 // библиотека fastlz
104 compression_fastlz,
105 };
106 // 0 - нет сжатия
107 // 1 - сжато fastlz
108
109 // Если есть сжатие то сжимается всё кроме bqMDLFileHeader
110 uint32_t m_compression = compression_null;
111
112 // если сжатие есть, размер не сжатых данных.
113 uint32_t m_uncmpSz = 0;
114 // если сжатие есть, размер сжатых данных.
115 uint32_t m_cmpSz = 0;
116
117 // 3D редактор имеет чуть иную систему координат.
118 // Верх не Y а Z. Что-бы не мучится с ручным
119 // преобразованием модели при экспорте, возни с
120 // позициями костей и т.д., проще прочитать всё как
121 // есть, и передать вращение. Потом в движке вращение
122 // будет использовано в умножение на матрицу кости,
123 // что приведёт к повороту модели, установки правильной
124 // ориентации.
125 // Не уверен на счёт моделей без кости. нужно проверить.
126 // Вращение в радианах.
127 float m_rotation[3] = { 0.f, 0.f, 0.f };
128
129 // Можно изменить масштаб. Так-же применяется как
130 // матрица, умножается на матрицу кости.
131 float m_scale = 1.f;
132
133 uint32_t m_reserved1 = 0;
134 uint32_t m_reserved2 = 0;
135 uint32_t m_reserved3 = 0;
136 uint32_t m_reserved4 = 0;
137};
