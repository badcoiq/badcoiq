37 size_t m_allocated;
38 void _reallocate(size_t new_capacity)
39 {
40 new_capacity += (size_t)ceilf((float)m_allocated * 0.5f);
41 auto tmp_size = new_capacity * sizeof(type);
42 pointer new_data = static_cast<type*>(bqMemory::malloc(tmp_size));
43 memset(new_data, 0, tmp_size);
44
45 if (m_data)
46 {
47 for (size_t i = 0u; i < m_size; ++i)
48 {
49 new(&new_data[i]) type(m_data[i]);
50 (&m_data[i])->~type();
51 }
52 bqMemory::free(m_data);
53 }
54 m_data = new_data;
55 m_allocated = new_capacity;
56 }

230 size_t i, j;
