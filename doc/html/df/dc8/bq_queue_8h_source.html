99 void Push(const Type& value)
100 {
101 if (m_memory->GetFreeCount())
102 {
103 _Node* newNode = (_Node*)m_memory->Allocate(1);
104 ::new(newNode) _Node();
105
106 newNode->m_data = value;
107
108 if (m_head)
109 {
110 m_tail->m_next = newNode;
111 m_tail = newNode;
112 }
113 else
114 {
115 m_head = newNode;
116 m_tail = newNode;
117 }
118 }
119 }
120
121 void Pop()
122 {
123 if (Size())
124 {
125 if (m_head)
126 {
127 _Node* popnode = m_head;
128
129 m_memory->FreeMemory(popnode);
130
131 m_head = m_head->m_next;
132
133 if (popnode == m_head)
134 {
135 m_head = 0;
136 m_tail = 0;
137 }
138 }
139 }
140 }
