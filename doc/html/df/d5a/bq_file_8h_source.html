55 BQ_ASSERT_ST(m_file);

194 bool Open(_open op, const char* path)
195 {
196 BQ_ASSERT_ST(!m_file);
197 BQ_ASSERT_ST(path);
198
199 if (m_file)
200 return false;
201
202 const char* mode_rb = "rb";
203 const char* mode_ab = "ab";
204 const char* mode_wb = "wb";
205
206 const char* mode = mode_rb;
207
208 switch (op)
209 {
210 case bqFile::_open::Open:
211 mode = mode_rb;
212 break;
213 case bqFile::_open::OpenAppend:
214 mode = mode_ab;
215 break;
216 case bqFile::_open::Create:
217 mode = mode_wb;
218 break;
219 default:
220 break;
221 }
222
223 fopen_s(&m_file, path, mode);
224
225 return true;
226 }
