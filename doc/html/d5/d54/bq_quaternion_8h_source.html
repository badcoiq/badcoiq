59 void Set(const bqVec4& v) { Set((float)v.x, (float)v.y, (float)v.z); }
60 void Set(float _x, float _y, float _z)
61 {
62 _x *= 0.5f;
63 _y *= 0.5f;
64 _z *= 0.5f;
65 float c1 = cos(_x);
66 float c2 = cos(_y);
67 float c3 = cos(_z);
68 float s1 = sin(_x);
69 float s2 = sin(_y);
70 float s3 = sin(_z);
71 w = (c1 * c2 * c3) + (s1 * s2 * s3);
72 x = (s1 * c2 * c3) - (c1 * s2 * s3);
73 y = (c1 * s2 * c3) + (s1 * c2 * s3);
74 z = (c1 * c2 * s3) - (s1 * s2 * c3);
75 }

106 bqVec3 RotateVector(const bqVec3& v)const
107 {
108 bqVec3 qv;
109 qv.x = -x;
110 qv.y = -y;
111 qv.z = -z;
112 float _w = w;
113
114 auto _d1 = qv.Dot(v);
115 auto _d2 = qv.Dot();
116 bqVec3 cr;
117 qv.Cross(v, cr);
118
119 bqVec3 rv;
120 rv.x = 2.f * _d1 * qv.x + (_w * _w - _d2) * v.x + 2.f * _w * cr.x;
121 rv.y = 2.f * _d1 * qv.y + (_w * _w - _d2) * v.y + 2.f * _w * cr.y;
122 rv.z = 2.f * _d1 * qv.z + (_w * _w - _d2) * v.z + 2.f * _w * cr.z;
123 return rv;
124 }
