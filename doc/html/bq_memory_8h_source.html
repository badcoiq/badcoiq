33// Использовать эти функции вместо malloc\free new\delete
34class bqMemory
35{
36public:
37 static void* malloc(size_t size);
38 static void* calloc(size_t size);
39 static void free(void*);
40 static void* realloc(void*, size_t size);
41};

43// Данные функции неудобно использовать для замены new и delete
44// Потому что для объектов классов нужно вызывать конструктор и деструктор.
45// Добавлю две функции которые будут выделять\освобождать память и вызывать конструктор\деструктор.
46template<typename _type, typename... Args>
47_type* bqCreate(Args&&... args)
48{
49 _type* ptr = (_type*)bqMemory::malloc(sizeof(_type));
50 if (ptr)
51 ::new(ptr) _type(std::forward<Args>(args)...); // вызов конструктора
52 return ptr;
53}
54
55template<typename _type>
56void bqDestroy(_type* ptr)
57{
58 assert(ptr);
59 ptr->~_type(); // вызов деструктора
60 bqMemory::free(ptr);
61}

67void operator delete(void* ptr){if (ptr) bqDestroy(ptr);}
