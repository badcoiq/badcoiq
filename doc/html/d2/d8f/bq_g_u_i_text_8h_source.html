64 virtual bqVec2f GetTextSize(const char32_t* text)
65 {
66 bqVec2f sz, position;
67 bqVec4f rct;
68 if (text)
69 {
70 if (text[0] != 0)
71 {
72 const char32_t* p = &text[0u];
73 while ((size_t)*p)
74 {
75 auto font = OnFont(Reason_default, *p);
76 bqGUIFontGlyph* g = font->GetGlyphMap()[*p];
77
78 rct.x = position.x;
79 rct.y = position.y;
80
81 rct.z = rct.x + (float)g->m_width;
82 rct.w = rct.y + (float)g->m_height;
83
84 position.x += float(g->m_width + g->m_overhang + g->m_underhang + font->m_characterSpacing);
85
86 switch (*p)
87 {
88 case U' ':
89 position.x += (float)font->m_spaceSize;
90 break;
91 case U'\t':
92 position.x += (float)font->m_tabSize;
93 break;
94 case U'\n':
95 position.y += (float)font->GetMaxSize().y;
96 position.x = 0.f;
97 break;
98 }
99
100 if (position.x > sz.x) sz.x = position.x;
101 if (position.y > sz.y) sz.y = position.y;
102 if ((float)g->m_height > sz.y) sz.y = (float)g->m_height;
103
104 ++p;
105 }
106 }
107 }
108 return sz;
109 }
