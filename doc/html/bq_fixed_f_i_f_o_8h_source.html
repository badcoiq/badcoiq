36template<typename Type, uint32_t size>
37class bqFixedFIFO
38{
39 uint32_t m_cur = 0;
40 uint32_t m_insert = 0;
41 uint32_t m_inUse = 0;
42 Type m_data[size];
43public:
44 bqFixedFIFO()
45 {
46 }
47
48 ~bqFixedFIFO()
49 {
50 }
51
52 void Push(const Type& v)
53 {
54 m_data[m_insert] = v;
55
56 if(m_inUse)
57 {
58 if(m_insert == m_cur)
59 {
60 ++m_cur;
61 if(m_cur >= size)
62 m_cur = 0;
63 }
64 }
65 else
66 {
67 m_cur = m_insert;
68 }
69
70 ++m_inUse;
71 if(m_inUse >= size)
72 m_inUse = size;
73
74 ++m_insert;
75 if(m_insert == size)
76 m_insert = 0;
77 }
78
79 bool TryPush(const Type& v)
80 {
81 if (m_inUse != size)
82 {
83 Push(v);
84 return true;
85 }
86 return false;
87 }
88
89 void Pop()
90 {
91 if (m_inUse)
92 {
93 --m_inUse;
94
95 ++m_cur;
96 if (m_cur >= size)
97 m_cur = 0;
98 }
99 }
100
101 Type& Front()
102 {
103 Type& r = m_data[m_cur];
104 return r;
105 }
106
107 bool IsEmpty() { return m_inUse == 0; }
108 void Clear() { m_inUse = 0; }
109 bool IsFull() { return m_inUse == size; }
110};
