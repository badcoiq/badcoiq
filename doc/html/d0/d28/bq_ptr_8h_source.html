33template<typename Type>
34class bqPtr
35{
36 using delete_method_type = void(bqPtr::*)();
37 delete_method_type m_delete = 0;
38
39 Type* m_ptr = 0;
40
41 void _deleteFree()
42 {
43 bqMemory::free(m_ptr);
44 }
45
46 void _deleteDestroy()
47 {
48 bqDestroy(m_ptr);
49 }
50
51public:
52
53 bqPtr(Type* p, delete_method_type delete_method)
54 :
55 m_ptr(p)
56 {
57 m_delete = delete_method;
58 }
59
60 ~bqPtr()
61 {
62 if (m_ptr)
63 (this->*m_delete)();
64 }
65
66 static delete_method_type Free()
67 {
68 return &bqPtr::_deleteFree;
69 }
70
71 static delete_method_type Destroy()
72 {
73 return &bqPtr::_deleteDestroy;
74 }
75
76 Type* Ptr() { return m_ptr; }
77
78 Type* Drop() { Type* p = m_ptr; m_ptr = 0; return p; }
79
80 Type* operator->() const { return m_ptr; }
81};
